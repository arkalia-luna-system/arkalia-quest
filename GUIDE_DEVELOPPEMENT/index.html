    from functools import wraps import time

    def cache_with_ttl(ttl_seconds=300): """D√©corateur de cache avec TTL""" def decorator(func): cache = {}

        @wraps(func)
    def wrapper(*args, **kwargs):
        # Cr√©er une cl√© unique
        key = str(args) + str(sorted(kwargs.items()))

        # V√©rifier le cache
        if key in cache:

        # Ex√©cuter la fonction
        result = func(*args, **kwargs)
        cache[key] = (result, time.time())

    return wrapper
return decorator
